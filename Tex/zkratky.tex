\chapter*{Seznam zkratek}
\addcontentsline{toc}{chapter}{Seznam zkratek}

\begin{sortedlist}
	\sortitem{API -- Application Programming Interface, rozhraní pro programování}
	\sortitem{GUI -- Graphical User Interface, grafické uživatelské rozhraní}
	\sortitem{KIV -- Katedra informatiky a~výpočetní techniky}
	\sortitem{OKS -- Ověřování kvality software}
	\sortitem{SQuaRE -- Software Quality Requirements and Evaluation}
	\sortitem{FURPS -- jeden z~nejčastějších modelů kvality software}
	\sortitem{UAT -- User Acceptance Testing, akceptační testování}
	\sortitem{UI -- User Interface, uživatelské rozhraní}
	\sortitem{CLI -- Command Line Interface, rozhraní příkazové řádky}
	\sortitem{OS -- Operating System, operační systém}
	\sortitem{IDE -- Integrated Development Environment, integrované vývojové prostředí}
	\sortitem{JRE -- Java Runtime Environment, prostředí pro běh aplikací napsaných v~programovacím jazyce Java}
	\sortitem{JDK -- Java Development Kit, vývojové nástroje pro programovací jazyk Java}
	\sortitem{OCR -- Optical Character Recognition, optické rozpoznání znaků}
	\sortitem{TS -- Test Suite, organizační jednotka seskupující testovací případy}
	\sortitem{TC -- Test Case, testovací případ}
\end{sortedlist}
